// Convert DaySlots to PlanDay lists format
function toPlanDayLists(daySlots: DaySlots): PlanDay["lists"] {
  const toMealArray = (meal?: SmartMeal) => meal ? [{
    id: meal.id,
    name: meal.name,
    servings: 1,
    ingredients: meal.ingredients?.map(ing => ({
      name: ing.item,
      quantity: ing.amount || "1",
      unit: ""
    })) || [],
    instructions: meal.instructions || [],
    nutrition: {
      calories: meal.macros.calories,
      protein: meal.macros.protein_g,
      carbs: meal.macros.carbs_g,
      fat: meal.macros.fat_g
    }
  }] : [];

  return {
    breakfast: toMealArray(daySlots.breakfast),
    lunch: toMealArray(daySlots.lunch),
    dinner: toMealArray(daySlots.dinner),
    snacks: toMealArray(daySlots.snack),
  };
}

// Create plan: All days get the same meals from Today's Plan
function makeDraftFromPlan(daySlots: DaySlots, daysCount: number): BuilderPlanJSON {
  const n = Math.min(7, Math.max(1, daysCount || 1));
  const dayLists = toPlanDayLists(daySlots);
  return {
    source: "diabetic",
    createdAtISO: new Date().toISOString(),
    days: Array.from({ length: n }, (_, i) => ({
      dayIndex: i,
      lists: structuredClone(dayLists),
    })),
  };
}

// Create empty plan with blank day slots (no meals)
function createEmptyPlan(daysCount: number): BuilderPlanJSON {
  const n = Math.min(7, Math.max(1, daysCount || 1));
  const emptyLists: PlanDay["lists"] = {
    breakfast: [],
    lunch: [],
    dinner: [],
    snacks: [],
  };
  return {
    source: "diabetic",
    createdAtISO: new Date().toISOString(),
    days: Array.from({ length: n }, (_, i) => ({
      dayIndex: i,
      lists: structuredClone(emptyLists),
    })),
  };
}


// ----- LocalMealCard Component -----
function LocalMealCard({
  slotLabel,
  meal,
  onRemove,
}: {
  slotLabel: string;
  meal: SmartMeal;
  onRemove: () => void;
}) {
  // Convert ingredients to strings if they're objects
  const ing = (meal.ingredients || []).map(i => {
    if (typeof i === 'string') return i;
    return `${i.amount} ${i.item}`;
  });
  const steps = meal.instructions || [];

  return (
    <Card className="bg-white/10 border-white/20 rounded-lg">
      <CardHeader className="p-3 pb-2">
        <div className="flex items-start justify-between">
          <div className="flex-1">
            <div className="text-white/70 text-xs uppercase mb-1">
              {slotLabel}
            </div>
            <CardTitle className="text-white text-base">{meal.name}</CardTitle>
          </div>
          <TrashButton
            onClick={onRemove}
            size="sm"
            confirm={true}
            confirmMessage={`Remove this ${slotLabel} from the plan?`}
            ariaLabel={`Remove ${slotLabel}`}
            title={`Remove ${slotLabel}`}
            data-testid={`button-remove-${slotLabel}`}
          />
        </div>
      </CardHeader>
      <CardContent className="p-3 pt-0 space-y-3">
        {/* Macros */}
        <div className="text-white/80 text-sm">
          <span className="font-bold">{meal.macros.calories} kcal</span> • P
          {meal.macros.protein_g}g C{meal.macros.carbs_g}g F{meal.macros.fat_g}g
        </div>

        {/* Health Badges Dropdown */}
        <HealthBadgesDropdown badges={meal.badges} className="mt-2" />

        {/* Add to Macros Button */}
        <MacroBridgeButton
          meal={{
            protein: meal.macros.protein_g || 0,
            carbs: meal.macros.carbs_g || 0,
            fat: meal.macros.fat_g || 0,
            calories: meal.macros.calories || 0,
            dateISO: new Date().toISOString().slice(0, 10),
            mealSlot: null,
            servings: 1,
          }}
          label="Add to Macros"
          source="diabetic-meal-builder"
        />

        {/* Ingredients */}
        {ing.length > 0 && (
          <div>
            <div className="font-semibold text-white text-sm mb-1">
              Ingredients
            </div>
            <ul className="text-white/80 text-sm list-disc pl-5 space-y-0.5">
              {ing.map((i, idx) => (
                <li key={idx}>{i}</li>
              ))}
            </ul>
          </div>
        )}

        {/* Instructions */}
        {steps.length > 0 && (
          <div>
            <div className="font-semibold text-white text-sm mb-1">
              Instructions
            </div>
            <ol className="text-white/80 text-sm list-decimal pl-5 space-y-0.5">
              {steps.map((s, idx) => (
                <li key={idx}>{s}</li>
              ))}
            </ol>
          </div>
        )}
      </CardContent>
    </Card>
  );
}

              <details className="mb-2" ref={lunchDetailsRef}>
                <summary className="cursor-pointer text-white/80">
                  Show options
                </summary>
                <div
                  key={`lunch-${UI_VERSION}`}
                  className="mt-3 w-full max-h-[60vh] overflow-y-auto pr-2 sm:max-h-none"
                  style={{ WebkitOverflowScrolling: "touch" }}
                >
                  {applyBglRange(diabeticMenuLunch).map((meal) => (
                    <div key={meal.id} className="flex w-full items-center justify-between gap-3 py-2 border-b border-white/10 last:border-0 max-w-full">
                      <div className="flex-1 min-w-0">
                        <div className="font-medium text-white">{meal.name}</div>
                        <div className="text-xs text-white/70">
                          {meal.macros.calories} kcal • P{meal.macros.protein_g}g C{meal.macros.carbs_g}g F{meal.macros.fat_g}g
                        </div>
                        {meal.badges && meal.badges.length > 0 && (
                          <div className="flex flex-wrap gap-1 mt-1">
                            {meal.badges.slice(0, 3).map((badge) => (
                              <span key={badge} className="px-2 py-0.5 text-xs bg-blue-600/20 text-blue-200 rounded">
                                {badge}
                              </span>
                            ))}
                          </div>
                        )}
                      </div>
                      <Button
                        size="sm"
                        onClick={() => setSlot("lunch", meal)}
                        className="shrink-0 bg-green-600 hover:bg-green-700 text-white"
                      >
                        Add
                      </Button>
                    </div>
                  ))}
                </div>
              </details>
            </CardContent>
          </Card>

          {/* Dinner */}
          <Card className={CARD_SHELL}>
            <CardHeader className="p-0 mb-2 flex items-center justify-between">
              <CardTitle className="text-white text-lg">Dinner</CardTitle>
              <AddOwnMealButton slot="dinner" onSave={(meal) => setSlot("dinner", meal)} variant="icon" />
            </CardHeader>
            <CardContent className="p-0">
              <details className="mb-2" ref={dinnerDetailsRef}>
                <summary className="cursor-pointer text-white/80">
                  Show options
                </summary>
                <div
                  key={`dinner-${UI_VERSION}`}
                  className="mt-3 w-full max-h-[60vh] overflow-y-auto pr-2 sm:max-h-none"
                  style={{ WebkitOverflowScrolling: "touch" }}
                >
                  {applyBglRange(diabeticMenuDinner).map((meal) => (
                    <div key={meal.id} className="flex w-full items-center justify-between gap-3 py-2 border-b border-white/10 last:border-0 max-w-full">
                      <div className="flex-1 min-w-0">
                        <div className="font-medium text-white">{meal.name}</div>
                        <div className="text-xs text-white/70">
                          {meal.macros.calories} kcal • P{meal.macros.protein_g}g C{meal.macros.carbs_g}g F{meal.macros.fat_g}g
                        </div>
                        {meal.badges && meal.badges.length > 0 && (
                          <div className="flex flex-wrap gap-1 mt-1">
                            {meal.badges.slice(0, 3).map((badge) => (
                              <span key={badge} className="px-2 py-0.5 text-xs bg-purple-600/20 text-purple-200 rounded">
                                {badge}
                              </span>
                            ))}
                          </div>
                        )}
                      </div>
                      <Button
                        size="sm"
                        onClick={() => setSlot("dinner", meal)}
                        className="shrink-0 bg-green-600 hover:bg-green-700 text-white"
                      >
                        Add
                      </Button>
                    </div>
                  ))}
                </div>
              </details>
            </CardContent>
          </Card>

          {/* Snacks */}
          <Card className={CARD_SHELL}>
            <CardHeader className="p-0 mb-2 flex items-center justify-between">
              <CardTitle className="text-white text-lg">Snacks</CardTitle>
              <AddOwnMealButton slot="snack" onSave={(meal) => setSlot("snack", meal)} variant="icon" />
            </CardHeader>
            <CardContent className="p-0">
              <details className="mb-2" ref={snackDetailsRef}>
                <summary className="cursor-pointer text-white/80">
                  Show options
                </summary>
                <div
                  key={`snack-${UI_VERSION}`}
                  className="mt-3 w-full max-h-[60vh] overflow-y-auto pr-2 sm:max-h-none"
                  style={{ WebkitOverflowScrolling: "touch" }}
                >
                  {applyBglRange(diabeticMenuSnacks).map((meal) => (
                    <div key={meal.id} className="flex w-full items-center justify-between gap-3 py-2 border-b border-white/10 last:border-0 max-w-full">
                      <div className="flex-1 min-w-0">
                        <div className="font-medium text-white">{meal.name}</div>
                        <div className="text-xs text-white/70">
                          {meal.macros.calories} kcal • P{meal.macros.protein_g}g C{meal.macros.carbs_g}g F{meal.macros.fat_g}g
                        </div>
                        {meal.badges && meal.badges.length > 0 && (
                          <div className="flex flex-wrap gap-1 mt-1">
                            {meal.badges.slice(0, 3).map((badge) => (
                              <span key={badge} className="px-2 py-0.5 text-xs bg-green-600/20 text-green-200 rounded">
                                {badge}
                              </span>
                            ))}
                          </div>
                        )}
                      </div>
                      <Button
                        size="sm"
                        onClick={() => setSlot("snack", meal)}
                        className="shrink-0 bg-green-600 hover:bg-green-700 text-white"
                      >
                        Add
                      </Button>
                    </div>
                  ))}
                </div>
              </details>
            </CardContent>
          </Card>
        </div>

        {/* Today's Plan */}
        <Card className="bg-black/80 backdrop-blur-xl border border-white/5 rounded-2xl shadow-2xl">
          <CardHeader className="p-4 mb-2 bg-black/20 backdrop-blur-lg border-b border-white/10">
            <CardTitle className="text-white/90 text-lg font-semibold flex items-center gap-2">
              <UtensilsCrossed className="h-5 w-5" />
              Today's Plan
            </CardTitle>
          </CardHeader>
          <CardContent className="grid gap-4 max-h-[56vh] overflow-auto sm:max-h-none">
            {plan.breakfast && (
              <LocalMealCard
                slotLabel="breakfast"
                meal={plan.breakfast}
                onRemove={() => removeSlot("breakfast")}
              />
            )}
            {plan.lunch && (
              <LocalMealCard
                slotLabel="lunch"
                meal={plan.lunch}
                onRemove={() => removeSlot("lunch")}
              />
            )}
            {plan.dinner && (
              <LocalMealCard
                slotLabel="dinner"
                meal={plan.dinner}
                onRemove={() => removeSlot("dinner")}
              />
            )}
            {plan.snack && (
              <LocalMealCard
                slotLabel="snack"
                meal={plan.snack}
                onRemove={() => removeSlot("snack")}
              />
            )}
            {!plan.breakfast && !plan.lunch && !plan.dinner && !plan.snack && (
              <div className="text-center py-8 text-white/60">
                <UtensilsCrossed className="h-8 w-8 mx-auto mb-2" />
                <p>
                  No meals selected yet. Choose meals from the options above to
                  build your plan.
                </p>
              </div>
            )}
          </CardContent>

          {/* MacroBridge Footer */}
          {(plan.breakfast || plan.lunch || plan.dinner || plan.snack) && (
            <div className="px-4 pb-4">
              <MacroBridgeFooter
                items={Object.values(plan).filter(Boolean).map(meal => ({
                  protein: meal.macros.protein_g,
                  carbs: meal.macros.carbs_g,
                  fat: meal.macros.fat_g,
                  calories: meal.macros.calories,
                }))}
                dateISO={new Date().toISOString().slice(0, 10)}
                mealSlot={null}
                variant="day"
                source="diabetic-meal-builder"
              />
            </div>
          )}
        </Card>
      </div>

      {/* Persistent Day Board */}
      <div className="mt-8">
        {/* Day Count Selector */}
        <GlassCard className="bg-black/60 backdrop-blur-lg border border-white/10 rounded-2xl p-4 mb-4">
          <div className="flex items-center gap-4">
            <Label className="text-white/80">Generate plan for:</Label>
            <Select value={daysCount.toString()} onValueChange={(v) => setDaysCount(+v)}>
              <SelectTrigger className="bg-black/50 text-white w-32">
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                {[1,2,3,4,5,6,7].map(n => (
                  <SelectItem key={n} value={n.toString()}>
                    {n} day{n>1?"s":""}
                  </SelectItem>
                ))}
              </SelectContent>
            </Select>
            <div className="text-sm text-white/60">
              Generate plan for 1-7 days.
            </div>
          </div>
        </GlassCard>

        <BuilderDayBoard 
          ref={builderDayBoardRef}
          builderKey="diabetic" 
          draft={draft}
          draftLabel="Create Menu"
          header="Your Diabetic Menu Plan"
        />
      </div>
      
      {/* Shopping Aggregate Bar */}
      <ShoppingAggregateBar
        ingredients={Object.values(plan).filter(Boolean).flatMap(meal => 
          normalizeIngredients(meal.ingredients || [])
        )}
        source="Diabetic Menu Builder"
      />
    </div>
  );
}
