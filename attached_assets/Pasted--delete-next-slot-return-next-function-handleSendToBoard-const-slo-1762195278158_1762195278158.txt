   delete next[slot];

  function handleSendToBoard() {
    const slots = ["breakfast", "lunch", "dinner", "snack"] as const;
    const meals = slots.map((s) => plan[s]).filter(Boolean) as SmartMeal[];
    if (!meals.length) {
      toast({
        title: "Nothing to send",
        description: "Pick meals first.",
        variant: "destructive",
      });
      return;
    }

    // Create board export
    const boardExport = {
      source: "diabetic-menu-builder",
      createdAt: new Date().toISOString(),
      mode: "maintain" as const,
      items: slots.map((slot) => {
        const meal = plan[slot];
        if (!meal) return null;
        return {
          slot,
          meal,
          portion: 1,
        };
      }).filter(Boolean),
    };

    // Save to localStorage
    localStorage.setItem("diabeticBoard:pendingImport", JSON.stringify(boardExport));

    toast({
      title: "Sent to Menu Board",
      description: `${meals.length} meals sent to your diabetic menu board.`,
    });

    // Navigate to board
    setLocation("/diabetic-menu-board");
  }
